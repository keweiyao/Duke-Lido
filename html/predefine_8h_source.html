157  // So that the log does not scream

159  stream << "0.000000E+00";
160  return stream;
161  }
162  int exponent = floor(log10(std::abs(x.value)));

164  // Transform here

167  if (base >= 0){
168  std::stringstream buff;

172  }

174  std::stringstream buff;

176  buff << std::fixed << base;
177  //Checking if we have a leading minus sign
178  std::string newbase = "-" + buff.str().substr(2, buff.str().size()-1);
179  stream << std::setw(8) << newbase;
180  }
181 
182  if (exponent >= 0) stream << "E+" << std::setw(2) << std::setfill('0') << exponent;
183  else stream << "E-" << std::setw(2) << std::setfill('0') << std::abs(exponent);
184  return stream;
