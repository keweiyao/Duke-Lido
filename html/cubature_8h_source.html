42 /* a vector integrand - evaluates the function at the given point x
43  (an array of length ndim) and returns the result in fval (an array
44  of length fdim). The void* parameter is there in case you have
45  to pass any additional data through to your function (it corresponds
46  to the fdata parameter you pass to cubature). Return 0 on
47  success or nonzero to terminate the integration. */
48 typedef int (*integrand) (unsigned ndim, const double *x, void *,
49  unsigned fdim, double *fval);
50 
51 /* a vector integrand of a vector of npt points: x[i*ndim + j] is the
52  j-th coordinate of the i-th point, and the k-th function evaluation
53  for the i-th point is returned in fval[i*fdim + k]. Return 0 on success
54  or nonzero to terminate the integration. */
55 typedef int (*integrand_v) (unsigned ndim, size_t npt,
56  const double *x, void *,
57  unsigned fdim, double *fval);

59 /* Different ways of measuring the absolute and relative error when
60  we have multiple integrands, given a vector e of error estimates
61  in the individual components of a vector v of integrands. These
62  are all equivalent when there is only a single integrand. */
63 typedef enum {
64  ERROR_INDIVIDUAL = 0, /* individual relerr criteria in each component */
65  ERROR_PAIRED, /* paired L2 norms of errors in each component,
66  mainly for integrating vectors of complex numbers */
67  ERROR_L2, /* abserr is L_2 norm |e|, and relerr is |e|/|v| */
68  ERROR_L1, /* abserr is L_1 norm |e|, and relerr is |e|/|v| */
69  ERROR_LINF /* abserr is L_\infty norm |e|, and relerr is |e|/|v| */
70 } error_norm;

72 /* Integrate the function f from xmin[dim] to xmax[dim], with at most
73  maxEval function evaluations (0 for no limit), until the given
74  absolute or relative error is achieved. val returns the integral,
75  and err returns the estimate for the absolute error in val; both
76  of these are arrays of length fdim, the dimension of the vector
77  integrand f(x). The return value of the function is 0 on success
78  and non-zero if there was an error. */
